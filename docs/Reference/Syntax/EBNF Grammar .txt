(* Stroscot EBNF Grammar - Literals *)

(* Program Structure *)
program = { module_declaration | import_statement | top_level_statement } ;

(* Headers, Modules, and Imports *)
module_declaration = "module" module_name [ docstring ] ;
import_statement = "import" module_name [ "as" identifier ] 
                 | "from" module_name "import" ( "*" | import_list ) ;
import_list = identifier { "," identifier } ;
module_name = identifier { "." identifier } ;

(* Top Level Statements *)
top_level_statement = class_definition 
                    | function_definition 
                    | variable_assignment 
                    | macro_definition 
                    | operator_fixity_declaration
                    | statement ;

(* Definitions, Declarations, and Assignments *)
class_definition = [ visibility_modifier ] "class" class_name [ "(" class_name { "," class_name } ")" ] 
                   [ docstring ] "{" { class_member } "}" ;

function_definition = [ visibility_modifier ] "func" function_name "(" parameter_list ")" 
                      [ ":" type_annotation ] [ docstring ] block ;

variable_assignment = [ visibility_modifier ] identifier [ ":" type_annotation ] "=" expression ;

parameter_list = [ parameter { "," parameter } ] ;
parameter = identifier [ ":" type_annotation ] [ "=" expression ] ;

class_member = function_definition | variable_assignment | special_method ;

(* Special Identifiers/Magic Methods *)
special_method = [ visibility_modifier ] "func" magic_method_name "(" parameter_list ")" 
                 [ ":" type_annotation ] [ docstring ] block ;
magic_method_name = "__" identifier "__" ;

(* Macros *)
macro_definition = "macro" identifier "(" parameter_list ")" [ docstring ] block ;

(* Operators & Fixity Declarations *)
operator_fixity_declaration = "fixity" ( "infix" | "prefix" | "postfix" ) number operator ;

(* Visibility/Access Modifiers *)
visibility_modifier = "public" | "private" | "protected" | "internal" ;

(* Control Flow *)
statement = expression_statement
          | if_statement
          | while_loop
          | for_loop
          | try_statement
          | context_statement
          | return_statement
          | break_statement
          | continue_statement
          | label_statement
          | block ;

if_statement = "if" expression block { "elif" expression block } [ "else" block ] ;
while_loop = "while" expression block ;
for_loop = "for" identifier "in" expression block ;
return_statement = "return" [ expression ] ;
break_statement = "break" [ label ] ;
continue_statement = "continue" [ label ] ;
expression_statement = expression ;

(* Labels *)
label_statement = label ":" statement ;
label = identifier ;

(* Exception Handling *)
try_statement = "try" block 
                { "catch" [ "(" identifier [ ":" type_annotation ] ")" ] block }
                [ "finally" block ] ;

(* Context Management *)
context_statement = "with" expression [ "as" identifier ] block ;

(* Block *)
block = "{" { statement } "}" ;

(* Expressions *)
expression = assignment_expression ;

assignment_expression = ternary_expression [ assignment_operator assignment_expression ] ;
assignment_operator = "=" | "+=" | "-=" | "*=" | "/=" | "%=" | "**=" | "//=" | "&=" | "|=" | "^=" | "<<=" | ">>=" ;

ternary_expression = logical_or_expression [ "?" expression ":" expression ] ;

logical_or_expression = logical_and_expression { "||" logical_and_expression } ;
logical_and_expression = equality_expression { "&&" equality_expression } ;
equality_expression = relational_expression { ( "==" | "!=" ) relational_expression } ;
relational_expression = bitwise_or_expression { ( "<" | ">" | "<=" | ">=" | "in" | "not" "in" ) bitwise_or_expression } ;

bitwise_or_expression = bitwise_xor_expression { "|" bitwise_xor_expression } ;
bitwise_xor_expression = bitwise_and_expression { "^" bitwise_and_expression } ;
bitwise_and_expression = shift_expression { "&" shift_expression } ;
shift_expression = additive_expression { ( "<<" | ">>" ) additive_expression } ;

additive_expression = multiplicative_expression { ( "+" | "-" ) multiplicative_expression } ;
multiplicative_expression = power_expression { ( "*" | "/" | "//" | "%" ) power_expression } ;
power_expression = unary_expression [ "**" power_expression ] ;

unary_expression = ( "+" | "-" | "!" | "~" | "not" ) unary_expression | postfix_expression ;

postfix_expression = primary_expression { postfix_operator } ;
postfix_operator = function_call | index_access | member_access ;

(* Function/Method Calls *)
function_call = "(" argument_list ")" ;
argument_list = [ expression { "," expression } ] ;

index_access = "[" expression { "," expression } "]" ;
member_access = "." identifier ;

primary_expression = literal
                   | variable_name
                   | function_literal
                   | "(" expression ")"
                   | collection_literal ;

(* Function/Lambda Literals *)
function_literal = "lambda" "(" parameter_list ")" [ ":" type_annotation ] "=>" expression
                 | "func" "(" parameter_list ")" [ ":" type_annotation ] block ;

(* Collection Literals *)
collection_literal = list_literal
                   | tuple_literal
                   | set_literal
                   | dictionary_literal
                   | matrix_literal ;

(* Lists/Arrays *)
list_literal = "[" [ expression { "," expression } [ "," ] ] "]" ;

(* Tuples *)
tuple_literal = "(" [ expression { "," expression } ] "," ")"
              | "(" expression "," { expression "," } ")" ;

(* Sets *)
set_literal = "{" [ expression { "," expression } [ "," ] ] "}" ;

(* Dictionaries/Maps/Objects *)
dictionary_literal = "{" [ dictionary_entry { "," dictionary_entry } [ "," ] ] "}" ;
dictionary_entry = expression ":" expression ;

(* Matrices *)
matrix_literal = "[" "[" expression { "," expression } "]" { "," "[" expression { "," expression } "]" } "]" ;

(* Literals *)
literal = number
        | string_literal
        | boolean
        | null_value
        | regular_expression
        | symbol ;

(* Numbers *)
number = integer | float | scientific_notation | hex_number | octal_number | binary_number ;
integer = digit { digit } ;
float = digit { digit } "." digit { digit } ;
scientific_notation = ( integer | float ) ( "e" | "E" ) [ "+" | "-" ] integer ;
hex_number = "0x" hex_digit { hex_digit } ;
octal_number = "0o" octal_digit { octal_digit } ;
binary_number = "0b" binary_digit { binary_digit } ;

(* Strings *)
string_literal = single_quoted_string
               | double_quoted_string
               | triple_quoted_string
               | interpolated_string ;

(* Single-Quoted Strings *)
single_quoted_string = "'" { single_string_char } "'" ;
single_string_char = ? any character except single quote and newline ? | escape_sequence ;

(* Double-Quoted Strings *)
double_quoted_string = "\"" { double_string_char } "\"" ;
double_string_char = ? any character except double quote and newline ? | escape_sequence ;

(* Triple-Quoted/Multi-line Strings *)
triple_quoted_string = "\"\"\"" { multi_line_char } "\"\"\""
                     | "'''" { multi_line_char } "'''" ;
multi_line_char = ? any character including newlines ? | escape_sequence ;

(* Interpolated/Templated Strings *)
interpolated_string = "f\"" { interpolated_char | interpolation } "\""
                    | "f'" { interpolated_char | interpolation } "'" ;
interpolated_char = ? any character except quote, newline, and { ? | escape_sequence ;
interpolation = "{" expression [ ":" format_spec ] "}" ;
format_spec = ? format specification string ? ;

escape_sequence = "\\" ( "n" | "t" | "r" | "\\" | "\"" | "'" | "0" | "x" hex_digit hex_digit | "u" hex_digit hex_digit hex_digit hex_digit ) ;

(* Booleans *)
boolean = "true" | "false" ;

(* Null/None Values *)
null_value = "null" | "none" | "nil" ;

(* Regular Expressions *)
regular_expression = "/" regex_body "/" [ regex_flags ] ;
regex_body = { regex_char } ;
regex_char = ? any character except / and newline ? | "\\" ? any character ? ;
regex_flags = ? combination of i, g, m, s, x flags ? ;

(* Symbols *)
symbol = ":" identifier ;

(* Other Semantic Keywords *)
semantic_keyword = "abstract" | "async" | "await" | "const" | "static" | "final" | "override" | "virtual" | "extern" ;

(* Comments *)
single_line_comment = "//" { ? any character except newline ? } ;
multi_line_comment = "/*" { ? any character ? } "*/" ;

(* Docstrings *)
docstring = triple_quoted_string ;

(* Identifiers *)
variable_name = identifier ;
function_name = identifier ;
class_name = upper_identifier ;
identifier = ( letter | "_" ) { letter | digit | "_" } ;
upper_identifier = upper_letter { letter | digit | "_" } ;

(* Type Annotations *)
type_annotation = class_name
                | generic_type
                | union_type
                | tuple_type
                | function_type ;

generic_type = class_name "<" type_annotation { "," type_annotation } ">" ;
union_type = type_annotation "|" type_annotation { "|" type_annotation } ;
tuple_type = "(" type_annotation { "," type_annotation } ")" ;
function_type = "(" [ type_annotation { "," type_annotation } ] ")" "=>" type_annotation ;

(* Operators *)
operator = "+" | "-" | "*" | "/" | "//" | "%" | "**" 
         | "==" | "!=" | "<" | ">" | "<=" | ">="
         | "&&" | "||" | "!" | "&" | "|" | "^" | "~" | "<<" | ">>"
         | "=" | "+=" | "-=" | "*=" | "/=" | "//=" | "%=" | "**="
         | "&=" | "|=" | "^=" | "<<=" | ">>="
         | "." | "->" | "?." | "??" | "..." ;

(* Character Classes *)
letter = ? a-z, A-Z, and Unicode letter characters ? ;
upper_letter = ? A-Z and Unicode uppercase letter characters ? ;
digit = ? 0-9 ? ;
hex_digit = ? 0-9, a-f, A-F ? ;
octal_digit = ? 0-7 ? ;
binary_digit = ? 0-1 ? ;

(* End of Grammar *)
